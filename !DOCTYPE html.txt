        function processClinicalWorkflow() {
            const hn = document.getElementById('patientHN').value.trim();
            const name = document.getElementById('patientName').value.trim() || 'ไม่ประสงค์ระบุชื่อ';
            let h = parseFloat(document.getElementById('height').value);
            let w = parseFloat(document.getElementById('weight').value);
            const formula = document.querySelector('input[name="bsaFormula"]:checked').value;
            const ampState = document.querySelector('input[name="amputationState"]:checked').value;
            const drug = document.getElementById('drugSelect').value;

            let calculatedWeight = w;
            let bsaAdjustmentFactor = 1.0;
            let ampSummaryText = "ไม่มีประวัติตัดแขนขา";

            if (ampState === 'amputee') {
                const ampLevel = document.getElementById('ampLevel').value;
                const ampMethod = document.getElementById('ampMethod').value;
                let wFactor = ampLevel === 'below_knee' ? 0.06 : 0.15;
                let bsaFactor = ampLevel === 'below_knee' ? 0.09 : 0.18;

                if (ampMethod === 'weight_method') {
                    calculatedWeight = w * (1 - wFactor);
                    ampSummaryText = `ตัดขา (${ampLevel === 'below_knee' ? 'BK' : 'AK'}) ปรับลดแบบ Weight Method`;
                } else {
                    bsaAdjustmentFactor = 1 - bsaFactor;
                    ampSummaryText = `ตัดขา (${ampLevel === 'below_knee' ? 'BK' : 'AK'}) ปรับลดแบบ BSA Method`;
                }
            }

            let bsa = 0;
            if (formula === 'mosteller') {
                bsa = Math.sqrt((h * calculatedWeight) / 3600);
            } else {
                bsa = 0.20247 * Math.pow((h / 100), 0.725) * Math.pow(calculatedWeight, 0.425);
            }
            bsa = bsa * bsaAdjustmentFactor;

            let dose = 0;
            let unit = "mg";
            let drugRuleText = "";

            if (drug === 'vincristine') {
                dose = bsa * 1.4;
                if (dose > 2.0) {
                    dose = 2.0;
                    drugRuleText = "ขนาดยาเกินเพดานมาตรฐาน ทำการจำกัดขนาดยา (Dose Cap) ไว้ที่สูงสุดไม่เกิน 2.0 mg อัตโนมัติเพื่อความปลอดภัย";
                } else {
                    drugRuleText = "คำนวณตามอัตราส่วนความสัมพันธ์มาตรฐานสากล (BSA × 1.4 mg)";
                }
            } else if (drug === 'carboplatin') {
                let targetAuc = parseFloat(document.getElementById('targetAuc').value) || 5;
                let gfr = parseFloat(document.getElementById('gfrValue').value);
                if (isNaN(gfr) || gfr < 0) {
                    alert('⚠️ กรุณาระบุค่า Clearance (GFR) เพื่อคำนวณสูตร Calvert ของตัวยา Carboplatin ค่ะ');
                    return;
                }
                let gfrCalc = gfr > 125 ? 125 : gfr;
                drugRuleText = gfr > 125 ? "ตรวจพบค่า GFR สูงเกินเกณฑ์มาตรฐาน ทำการจำกัดเพดาน GFR ไว้สูงสุดที่ 125 ml/min" : "ประมวลผลผ่าน Calvert Equation เรียบร้อย";
                dose = targetAuc * (gfrCalc + 25);
            } else if (drug === 'bleomycin') {
                dose = 30;
                unit = "units";
                drugRuleText = "ขนาดยาในสูตรการรักษาเฉพาะทาง (BEP Regimen) กำหนดขนาดยาคงที่ (Fixed Dose) ที่ 30 units";
            }

            const jsonObject = {
                id: Date.now(), // สร้าง ID เพื่อใช้จับทำ Animation แถวใหม่
                timestamp: new Date().toLocaleTimeString('th-TH'),
                hn: hn,
                patientName: name,
                formulaUsed: formula === 'mosteller' ? 'Mosteller' : 'DuBois',
                calculatedBsaM2: parseFloat(bsa.toFixed(4)),
                prescribedDose: `${dose.toFixed(2)} ${unit}`,
                clinicalNote: drugRuleText
            };

            let archivedLogs = JSON.parse(localStorage.getItem("clinicalDoseHistoryLogs")) || [];
            archivedLogs.unshift(jsonObject);
            localStorage.setItem("clinicalDoseHistoryLogs", JSON.stringify(archivedLogs));

            // 🎬 ใส่เอฟเฟกต์ตัวเลขเด้งยอดฮิต (Pop Counter)
            document.getElementById('bsaDisplayValue').innerText = bsa.toFixed(4);
            document.getElementById('doseDisplayValue').innerText = dose.toFixed(2);
            document.getElementById('doseUnitValue').innerText = unit;
            
            const bsaBox = document.getElementById('bsaDisplayContainer');
            const doseBox = document.getElementById('doseDisplayContainer');
            bsaBox.classList.add('animate-pop');
            doseBox.classList.add('animate-pop');
            setTimeout(() => {
                bsaBox.classList.remove('animate-pop');
                doseBox.classList.remove('animate-pop');
            }, 400);

            document.getElementById('resMethod').innerText = formula === 'mosteller' ? 'Mosteller' : 'DuBois';
            document.getElementById('resAmp').innerText = ampSummaryText;
            document.getElementById('resDrugRule').innerText = drugRuleText;

            // 🎬 แสดงกล่องสรุปพร้อมอนิเมชันสไลด์ขึ้นมา
            const resultBox = document.getElementById('hospitalResultBox');
            resultBox.classList.remove('hidden');
            setTimeout(() => {
                resultBox.classList.replace('scale-95', 'scale-100');
                resultBox.classList.replace('opacity-0', 'opacity-100');
            }, 50);

            renderHistoryLogs(jsonObject.id);
        }

        function renderHistoryLogs(newestId = null) {
            const logs = JSON.parse(localStorage.getItem("clinicalDoseHistoryLogs")) || [];
            const tbody = document.getElementById("historyLogTableBody");
            
            if (logs.length === 0) {
                tbody.innerHTML = `<tr><td colspan="6" class="p-6 text-center text-slate-500 font-mono text-lg font-bold">No data logged in this browser session.</td></tr>`;
                return;
            }

            tbody.innerHTML = logs.map(item => {
                // 🎬 ถ้าเป็นแถวล่าสุดที่เพิ่งกดคำนวณสดๆ ร้อนๆ ให้เพิ่มคลาสแอนิเมชันสไลด์เฟดเข้ามา
                const animationClass = (newestId && item.id === newestId) ? 'animate-row-in' : '';
                return `
                    <tr class="border-b border-slate-700 hover:bg-slate-900/60 transition-colors ${animationClass}">
